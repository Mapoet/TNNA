\begin{frame}

$$\frac{d^n F(t,X_t,Y_t)}{d t^n}=F_n(t,X_t,Y_t)$$
$$F(t,X_t,Y_t)=F(t_0,X_{t_0},Y_{t_0})+\sum_{n=1}^{\infty}{\frac{F_n(t_0,X_{t_0},Y_{t_0})}{n!}(t-t_0)^n}$$
$$Y_t - Y_{t_0}=F(t_0,X_{t_0},Y_{t_0}){\delta t}$$

\end{frame}
